myApp.run([
  "$rootScope",
  "$transitions",
  function ($rootScope, $transitions) {
      $rootScope.navigateSlide = "";
      const isOctober = new Date().getMonth() === 9;
      // to disable maintenance set var to false and rename app_offline.html
      const isMaintenanceMode = false;
      $rootScope.bannerDesktop = isOctober ? "BannerImage_1000418_OctoberBCA_newsiteBCA2.png" : "LG_PNG_Banner_Images-02.png";
      $rootScope.bannerMobile = isOctober ? "BannerImage_100918_OctoberBCA_mobile_version.png" : "LG_PNG_Banner_Images-01.png";

      $transitions.onStart({}, function (trans) {
          if (isMaintenanceMode && trans.to().name !== 'maintenance') {
              return trans.router.stateService.target('maintenance');
          }
      });

    $transitions.onBefore({}, function () {
      $rootScope.navigateSlide = $rootScope.navigateSlide || "navigate-slide";
      window.scrollTo(0, 0);
    });

    $transitions.onSuccess({}, function () {
        removeModalBackdrop();
    })

    // Function to remove the modal backdrop
    function removeModalBackdrop() {
        var backdrop = document.querySelector('.modal-backdrop');
        if (backdrop) {
            backdrop.remove();
        }
    }
  },
]);

myApp.filter('rangeToCurrentYear', function() {
  return function(result, start, shouldReverse) {
      const startParsed = parseInt(start) || 2005;
      const currentYear = new Date().getFullYear()
      for (let i = startParsed; i <= currentYear; i++) {
          result.push(i);
      }
      return shouldReverse ? result.reverse() : result;
  };
});

myApp.directive('phoneMask', function () {
  return {
    restrict: 'A',
    require: 'ngModel',
    link: function (scope, element, attrs, ngModelCtrl) {
      ngModelCtrl.$parsers.push(function (value) {
        if (!value) return '';

        const digits = value.replace(/\D/g, '').substring(0, 10); // limit to 10 digits
        let formatted = '';

        if (digits.length > 6) {
          formatted = `(${digits.slice(0, 3)}) ${digits.slice(3, 6)}-${digits.slice(6)}`;
        } else if (digits.length > 3) {
          formatted = `(${digits.slice(0, 3)}) ${digits.slice(3)}`;
        } else if (digits.length > 0) {
          formatted = `(${digits}`;
        }

        ngModelCtrl.$setViewValue(formatted);
        ngModelCtrl.$render();
        return formatted;
      });
    }
  };
});

myApp.directive('dateRangeValidator', function() {
  return {
    require: 'ngModel',
    scope: {
      companionDate: '=',   // The other date to compare with
      comparison: '@'       // 'beforeOrEqual' or 'afterOrEqual'
    },
    link: function(scope, element, attrs, ngModel) {
      function validate(currentDate, companionDate) {
        if (!currentDate || !companionDate) return true;

        const current = new Date(currentDate);
        const companion = new Date(companionDate);

        if (scope.comparison === 'beforeOrEqual') {
          return current <= companion;
        } else if (scope.comparison === 'afterOrEqual') {
          return current >= companion;
        }
        return true;
      }

      ngModel.$validators.dateRange = function(modelValue) {
        return validate(modelValue, scope.companionDate);
      };

      scope.$watch('companionDate', function() {
        ngModel.$validate();
      });
    }
  };
});

myApp.directive('currencyInput', ['$filter', function($filter) {
  return {
    restrict: 'A',
    require: 'ngModel',
    link: function(scope, element, attrs, ngModelCtrl) {
      const currencySymbol = attrs.currencySymbol || '$';

      function formatView(value) {
        if (!value || isNaN(value)) {
          return '';
        }
        return $filter('currency')(value, currencySymbol, 2);
      }

      function parseView(value) {
        if (!value) { 
          ngModelCtrl.$setValidity('currencyValid', false);
          return null;
        }

        const numeric = parseFloat(value.toString().replace(/[^0-9.-]+/g, ''));
        const isValid = !isNaN(numeric);
        ngModelCtrl.$setValidity('currencyValid', isValid);

        return isValid ? numeric : null;
      }

      ngModelCtrl.$formatters.push(formatView);
      ngModelCtrl.$parsers.push(parseView);

      // Format on blur (when user leaves the input)
      element.on('blur', function () {
        const value = ngModelCtrl.$modelValue;
        element.val(formatView(value));
      });

      // Show raw value on focus
      element.on('focus', function () {
        const value = ngModelCtrl.$modelValue;
        element.val(value != null ? value.toString() : '');
      });

      // Format view initially
      scope.$watch(() => ngModelCtrl.$modelValue, function (newVal) {
        const isValid = typeof newVal === 'number' && !isNaN(newVal);
        ngModelCtrl.$setValidity('currencyValid', isValid);

        if (document.activeElement !== element[0]) {
          element.val(formatView(newVal));
        }
      });
    }
  };
}]);
