    function bindDropdownEvents(shadowRoot) {
      const buttons = shadowRoot.querySelectorAll(".toggle-btn");

      buttons.forEach((button) => {
        button.addEventListener("click", (event) => {

          event.stopPropagation(); // Prevent event from bubbling

          const panelId = button.getAttribute("aria-controls");
          const panel = shadowRoot.getElementById(panelId);
          const isExpanded = button.getAttribute("aria-expanded") === "true";

          if (!isExpanded) {
            closeAllPanels(shadowRoot, button);
            button.setAttribute("aria-expanded", "true");
            expandPanel(panel);
          } else {
            button.setAttribute("aria-expanded", "false");
            collapsePanel(panel);
          }
        });
      });
    }

    // Utility functions that operate inside the shadow DOM
    function closeAllPanels(shadowRoot, except = null) {
      shadowRoot.querySelectorAll(".toggle-btn").forEach((btn) => {
        const panelId = btn.getAttribute("aria-controls");
        const panel = shadowRoot.getElementById(panelId);

        if (btn !== except && panel?.classList.contains("open")) {
          btn.setAttribute("aria-expanded", "false");
          collapsePanel(panel);
        }
      });
    }

    function expandPanel(panel) {
      panel.hidden = false;
      panel.classList.add("open");
      panel.style.height = "0px";
      requestAnimationFrame(() => {
        const fullHeight = panel.scrollHeight + "px";
        panel.style.height = fullHeight;
        panel.addEventListener("transitionend", function handler() {
          panel.style.height = "auto";
          panel.removeEventListener("transitionend", handler);
        });
      });
    }

    function collapsePanel(panel) {
      const currentHeight = panel.scrollHeight + "px";
      panel.style.height = currentHeight;
      panel.offsetHeight;
      panel.style.height = "0px";
      panel.addEventListener("transitionend", function handler() {
        panel.hidden = true;
        panel.classList.remove("open");
        panel.style.height = "";
        panel.removeEventListener("transitionend", handler);
      });
    }

        const translateEl = document.getElementById("google_translate_element");

        if (!translateEl) {
          console.warn("Google Translate element not found in DOM.");
          return;
        }

        translateEl.click();

        setTimeout(() => {
          const iframes = document.querySelectorAll("iframe.skiptranslate");
          let targetIframe = null;

          for (const iframe of iframes) {
            if (!iframe.parentElement.classList.contains("skiptranslate")) {
              targetIframe = iframe;
              break;
            }
          }

          if (targetIframe) {
            const doc =
              targetIframe.contentDocument ||
              targetIframe.contentWindow.document;
            const menuItems = doc.querySelectorAll("a .text");

            for (const item of menuItems) {
              if (
                item.innerText.trim().toLowerCase() ===
                selectedLanguage.trim().toLowerCase()
              ) {
                item.parentElement.click();
                return;
              }
            }
          } else {
            console.error("Google Translate iframe not found.");
          }
        }, 500);

          if (targetIframe) {
            const doc =
              targetIframe.contentDocument ||
              targetIframe.contentWindow.document;
            const menuItems = doc.querySelectorAll("a .text");

            for (const item of menuItems) {
              if (
                item.innerText.trim().toLowerCase() ===
                selectedLanguage.trim().toLowerCase()
              ) {
                item.parentElement.click();
                return;
              }
            }
          } else {
            console.error("Google Translate iframe not found.");
          }
        }, 500);
      });

      /**************************
        SEARCH OVERLAY
      **************************/

      // Search overlay logic
      const searchButton = root.getElementById("search-button");
      const closeButton = root.getElementById(
        "mdc-custom_fullscreen-close-button"
      );
      const searchOverlay = root.getElementById("mdc-custom_search-overlay");
      const searchInput = root.querySelector(
        ".mdc-custom_fullscreen-search-input"
      );
      const searchForm = root.querySelector(
        ".mdc-custom_fullscreen-searchform"
      );

      if (
        searchButton &&
        closeButton &&
        searchOverlay &&
        searchInput &&
        searchForm
      ) {
        searchButton.addEventListener("click", () => {
          searchOverlay.classList.add("show");
          setTimeout(() => searchInput.focus(), 200);
        });

        closeButton.addEventListener("click", () => {
          searchOverlay.classList.remove("show");
        });

        const alignSearchForm = () => {
          searchForm.style.marginTop = `${
            window.innerHeight / 2 - searchForm.offsetHeight / 2
          }px`;
        };
        alignSearchForm();
        window.addEventListener("resize", alignSearchForm);
      }
    }

                  const normalized = href.replace(/^\/+/, ""); // remove leading slashes

                }
              });


              return wrapper.innerHTML;

              if (!path) return "#";

              // If it's already a full URL, return as-is
              if (path.startsWith("http")) return path;

              // Remove leading slashes
              const cleanPath = path.replace(/^\/+/, "");

              // If it already has an extension (like .png), don't append .page

              return hasExtension

            });
